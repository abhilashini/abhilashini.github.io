func ProcessTasks(tasks []string) {
  ch := make(chan string)
  for _, task := range tasks {
    go func(t string) { 
      ch <- fmt.Sprintf("Processed: %s", t) 
    }(task)
  }
  for range tasks { fmt.Println(<-ch) }
}
      

# Context Management
@contextmanager
def atomic_write(path):
    tmp = f"{path}.tmp"
    with open(tmp, 'w') as f:
        yield f
    os.replace(tmp, path)
